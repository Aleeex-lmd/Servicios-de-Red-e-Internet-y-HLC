h3. Deployment.yml

<pre><code=yml>
apiVersion: apps/v1
kind: Deployment
metadata:
  name: web-deployment
spec:
  replicas: 2
  selector:
    matchLabels:
      app: test-web
  template:
    metadata:
      labels:
        app: test-web
    spec:
      containers:
        - name: test-web-container
          image: iesgn/test_web:latest
          ports:
            - containerPort: 80
</code></pre>
